<style>
  .pencil-toast {
    z-index: 30;
  }
</style>

<pc-button @click="autoClose">设置关闭时间</pc-button>
<pc-button @click="onClickButton">上方弹出</pc-button>

methods: {
  autoClose() {
    this.$toast('3秒后关闭', { autoClose: 3 });
  },
  onClickButton() {
    this.$toast("点击弹出提示", {
      closeButton: {
        text: "确定",
        callback: () => {
          this.date = new Date();
          console.log("To do something..."); //To do something...
        }
      }
    });
  }
}
